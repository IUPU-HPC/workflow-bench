Install transport methods¶

Dataspaces/Dimes¶

DataSpaces/Dimes 1.6.2 is used by us.

wget http://personal.cac.rutgers.edu/TASSL/projects/data/downloads/dataspaces-latest.tar.gz

Note

This may lead to newer versions than 1.6.2, which we may not have tested

DataSpaces can be configured using the scripts:

#!/bin/bash
#
# This script is for configuring adios on the authors' machines
# You can study it to figure out how to configure adios on your system
#

#SRCDIR=`dirname ${BASH_SOURCE[0]}`

if [ `hostname | cut -c 1-2` == "br" ]; then
    #######################
    # Bridges #
    #######################
    echo "Configure on  Bridges."
    #module load gcc
    #module load mpi/gcc_mvapich
    #module load cmake
    #rdmacm_root=~/envs/rdmacm/
    #CFLAGS="-DLINUX -g -O3 -fPIC -DHAVE_INFINIBAND" \

    # if tau is configured 
    if [[ $1 == tau ]]; then
        echo "using tau compiler and mpi"
        myCC=tau_cc.sh
        myCXX=tau_cxx.sh
        myFC=tau_f90.sh
        myprefix=${HOME}/envs/Dataspacesroot_tau

    # if use intel compiler/mpi
    elif [[ $1 == intel ]]; then
             echo "using intel compiler and mpi"
        myCC=mpiicc
        myCXX=mpiicpc
        myFC=mpiifort
        myprefix=${HOME}/envs/Dataspacesroot
        

    else
       
        echo "using intel compiler with mvapich"
        myCC="mpicc"
        myCXX="mpicxx"
        myFC="mpif90"
        myCFLAGS=""
        myFCFLAGS=""
        myprefix=${HOME}/envs/gcc_mvapich/Dataspacesroot


    fi


    module list

    #export CFLAGS="-I /usr/src/kernels/3.10.0-514.26.2.el7.x86_64/include/ -g -O3 -fPIC"
    #export LIBS=" -lpthread -lm"
    #${SRCDIR}/configure --prefix=${HOME}/envs/Dataspacesroot \
    ./configure --prefix=${myprefix} \
        CC=${myCC} \
        CXX=${myCXX} \
        FC=${myFC} \
        CFLAGS="${myCFLAGS} -O3" \
        FCFLAGS="${myFCFLAGS}" \
        LIBS="-lpthread -lm" \
        --enable-dimes \
        --with-ib-interface=ib0 \
        --with-dimes-rdma-buffer-size=1024 \

elif [ `hostname | cut -c 8-16` == "stampede2" ]; then
    #######################
    # Stampede #
    #######################
    echo "Configure on  stampede2"
    # if tau is configured 
    if [[ $1 == tau ]]; then
        myCC=tau_cc.sh
        myCXX=tau_cxx.sh
        myFC=tau_f90.sh
        myprefix=${WORK}/envs/Dataspacesroot_tau
    elif [[ $1 == mvapich ]]; then
        myCC=mpicc
        myCXX=mpicxx
        myFC=mpif90
        myprefix=${WORK}/envs/Dataspacesroot_mvapich

    else
        myCC=mpicc
        myCXX=mpicxx
        myFC=mpif90
        myCFLAGS="-xCORE-AVX2 -axMIC-AVX512"
        myprefix=${WORK}/envs/Dataspacesroot
    fi


    module list

    #export CFLAGS="-I /usr/src/kernels/3.10.0-514.26.2.el7.x86_64/include/ -g -O3 -fPIC"
    #export LIBS=" -lpthread -lm"
    #${SRCDIR}/configure --prefix=${HOME}/envs/Dataspacesroot \
    ./configure --prefix=${myprefix} \
        CC=${myCC} \
        CXX=${myCXX} \
        FC=${myFC} \
        CFLAGS="${myCFLAGS} -O3" \
        LIBS="-lpthread -lm" \
        --enable-dimes \
        --with-dimes-rdma-buffer-size=1024 \
        --with-ib-interface=ib0 \
        --with-infiniband-timeout=300 \

 


elif [ `hostname | cut -c 1-5` == "comet" ]; then

    #######################
    # Comet #
    #######################
    echo "Configure on  Comet"
    module purge
    module load gnutools
    module load gnu
    module load mvapich2_ib/2.1
    module load cmake

    export CC=mpicc
    export CXX=mpicxx
    export FC=mpif90

    export CFLAGS="-g -O3 -fPIC"
    export LIBS=" -lpthread -lm"
    ${SRCDIR}/configure --prefix=${HOME}/envs/Dataspacesroot \
        --disable-fortran \
        --enable-dimes \

else
    echo "Could not determine what machine is this."
    echo "This script is for configuring adios on the authors' machines."
    echo "You can study it to figure out how to configure adios on your system."
fi

ADIOS¶

ADIOS can be obtained by

wget wget http://users.nccs.gov/~pnorbert/adios-1.13.0.tar.gz

It can be configured using the script:

#!/bin/bash
#
# This script is for configuring adios on the authors' machines
# You can study it to figure out how to configure adios on your system
#

SRCDIR=`dirname ${BASH_SOURCE[0]}`

if [ `hostname | cut -c 1-2` == "br" ]; then

    #######################
    # Bridges             #
    #######################
    echo "Configure on ADIOS On Bridges."


    if [[ $1 == tau ]]; then
        echo "using tau compiler and mpi"
        #myCC=tau_cc.sh
        #myCXX=tau_cxx.sh
        #myFC=tau_f90.sh
        # run with make CC=tau_cc.sh
        ## tau  is instructed with gcc, instaled by Feng Li in env
        myCC=tau_cc.sh
        myCXX=tau_cxx.sh
        myFC=tau_f90.sh
        myDataspaces=${HOME}/envs/Dataspacesroot_tau
        myFlexpath=${HOME}/envs/Flexpath_gcc_mvapich_tau/
        myprefix=${HOME}/envs/ADIOSroot_gcc_mvapich_tau

    # if use intel compiler/mpi
    elif [[ $1 == intel ]]; then
        echo "using intel compiler and mpi"
        myCC=mpiicc
        myCXX=mpiicpc
        myFC=mpiifort
        myDataspaces=${HOME}/envs/Dataspacesroot
        myFlexpath=${HOME}/envs/Flexpath
        myprefix=${HOME}/envs/ADIOSroot


        
    else
        echo "using gnu compiler with mvapich"
        myCC=mpicc
        myCXX=mpicxx
        myFC=mpif90
        myDataspaces=${HOME}/envs/gcc_mvapich/Dataspacesroot
        myFlexpath=${HOME}/envs/gcc_mvapich/Flexpath
        myprefix=${HOME}/envs/gcc_mvapich/ADIOSroot

        
    fi
    module list

    #rdmacm_root=~/envs/rdmacm/

    ./configure --prefix=${myprefix} \
        CC=${myCC} \
        CXX=${myCXX} \
        FC=${myFC} \
        MPICC=${myCC} \
        MPICXX=${myCXX} \
        MPIFC=${myFC} \
        CFLAGS="-O3 -fPIC"\
        LIBS="-lrdmacm -libverbs -lpthread -lm" \
        --enable-dependency-tracking \
        --enable-timers \
        --disable-fortran \
        --disable-timer-events \
        --with-dataspaces=${myDataspaces} \
        --with-dimes=${myDataspaces}  \
        --with-flexpath=${myFlexpath} \

elif [ `hostname | cut -c 8-16` == "stampede2" ]; then
    #######################
    # Stampede #
    #######################
    echo "Configure on  stampede2"
    
    if [[ $1 == tau ]]; then
        myCC=tau_cc.sh
        myCXX=tau_cxx.sh
        myFC=tau_f90.sh
        myDataspaces=${WORK}/envs/Dataspacesroot_tau
        myprefix=${WORK}/envs/ADIOSroot_tau
    else
        myCC=mpicc
        myCXX=mpicxx
        myFC=mpif90
        myCFLAGS="-xCORE-AVX2 -axMIC-AVX512"
        myDataspaces=${WORK}/envs/Dataspacesroot
        myprefix=${WORK}/envs/ADIOSroot
    fi
    module list

    #rdmacm_root=~/envs/rdmacm/

    ./configure --prefix=${myprefix} \
        CC=${myCC} \
        CXX=${myCXX} \
        FC=${myFC} \
        MPICC=${myCC} \
        MPICXX=${myCXX} \
        MPIFC=${myFC} \
        CFLAGS="-O3 -fPIC ${myCFLAGS}" \
        LIBS="-lrdmacm -libverbs -lpthread -lm" \
        --enable-dependency-tracking \
        --enable-timers \
        --disable-fortran \
        --disable-timer-events \
        --with-dataspaces=${myDataspaces} \
        --with-dimes=${myDataspaces}  \
        --with-flexpath=${WORK}/envs/Flexpath \



elif [ `hostname | cut -c 1-5` == "comet" ]; then

    #######################
    # Comet        #
    #######################
    echo "Configure ADIOS on Comet."
    module purge
    module load gnutools/2.69
    module load gnu mvapich2_ib
    module load cmake
    #module load gnutools
    #module load gnu openmpi_ib
    #module load cmake


    export MPICC=mpicc
    export MPICXX=mpicxx
    export MPIFC=mpif90
    export CC=mpicc
    export CXX=mpicxx
    export FC=mpif90

    export CFLAGS="-g -O3 -fPIC  "
    export LIBS="-lrdmacm -libverbs -lpthread -lm"
    ${SRCDIR}/configure --prefix=${HOME}/envs/ADIOSroot \
        --enable-dependency-tracking \
        --enable-timers \
        --disable-timer-events \
        --disable-fortran \
        --with-dataspaces=${HOME}/envs/Dataspacesroot  \
        --with-dimes=${HOME}/envs/Dataspacesroot  \
        --with-flexpath=${HOME}/envs/Flexpath \

else
    echo "Could not determine what machine is this."
    echo "This script is for configuring adios on the authors' machines."
    echo "You can study it to figure out how to configure adios on your system."
fi

Flexpath¶

First obtain the package

mkdir flexpath_adios_1_13 && cd flexpath_adios_1_13
wget http://www.cc.gatech.edu/systems/projects/EVPath/chaos_bootstrap.pl

Bootstrap:

perl chaos_bootstrap.pl adios-1.13 $WORK/envs/Flexpath &> config.lifen.log &

build and install:

perl chaos_build.pl  -c ~/Workspaces/General_Data_Broker/lbm_adios/configs/Flexpath_build_config &> make.log &

Decaf¶

We use decaf, commit number: 637eb58 The decaf project is here

Users can follow the instructions to compelete installation. We also provide installation scripts for bridges:

cmake .. \
-DCMAKE_INSTALL_PREFIX=$WORK/software/install \
-DCMAKE_C_COMPILER=gcc \
-DCMAKE_CXX_COMPILER=g++ \
-DCMAKE_CXX_FLAGS=-O3 \
-DCMAKE_CXX_FLAGS=-O3 \
-Dtransport_mpi=on \
-Dtransport_cci=on \
-Dtransport_file=off \
-DCCI_PREFIX:PATH=$WORK/software/install \
-DBOOST_ROOT:PATH=$WORK/software/install

and for stampede2:

cmake .. \
-DCMAKE_INSTALL_PREFIX=/work/04446/tg837458/stampede2/software/install\
-DCMAKE_C_COMPILER=icc \
-DCMAKE_CXX_COMPILER=icpc \
-Dtransport_mpi=on \
-Dtransport_cci=on \
-Dtransport_file=off \
-DCCI_PREFIX:PATH=/work/04446/tg837458/stampede2/software/install
